class RemoteDataSource {
  final Dio _dio;

  RemoteDataSource(this._dio);

  Future<List<ItemModel>> getItems() async {
    try {
      final response = await _dio.get('/items');
      return (response.data as List)
          .map((json) => ItemModel.fromJson(json))
          .toList();
    } on DioException catch (e) {
      throw ServerException(e.message ?? 'Server error');
    }
  }
}

class LocalDataSource {
  late Box<ItemModel> _itemsBox;

  Future<void> init() async {
    _itemsBox = await Hive.openBox<ItemModel>('items');
  }

  Future<void> cacheItems(List<ItemModel> items) async {
    await _itemsBox.clear();
    await _itemsBox.addAll(items);
  }

  List<ItemModel> getCachedItems() {
    return _itemsBox.values.toList();
  }
}

class CacheDataSource {

  }

    return _prefs.getString(key);
  }

  Future<void> clearCache() async {
    await _prefs.clear();
