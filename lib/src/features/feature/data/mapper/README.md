class EntityMapper {
  ItemEntity mapModelToEntity(ItemModel model) {
    return ItemEntity(
      id: model.id ?? throw MappingException('Item id cannot be null'),
      name: model.name ?? 'Unnamed Item',
      price: model.price ?? 0.0,
      createdAt: DateTime.parse(model.createdAt),
    );
  }

  List<ItemEntity> mapModelListToEntityList(List<ItemModel> models) {
    return models.map(mapModelToEntity).toList();
  }
}

// model_mapper.dart  
class ModelMapper {
  ItemModel mapEntityToModel(ItemEntity entity) {
    return ItemModel(
      id: entity.id,
      name: entity.name,
      price: entity.price,
      createdAt: entity.createdAt.toIso8601String(),
    );
  }

  List<ItemModel> mapEntityListToModelList(List<ItemEntity> entities) {
    return entities.map(mapEntityToModel).toList();
  }
}
