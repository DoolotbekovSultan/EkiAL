class Bloc extends Bloc<BlocEvent, BlocState> {
  final GetItemsUseCase _getItemsUseCase;

  Bloc(this._getItemsUseCase) : super(const BlocState.initial()) {
    on<LoadItemsEvent>(_onLoadItems);
  }

  Future<void> _onLoadItems(
    LoadItemsEvent event,
    Emitter<BlocState> emit,
  ) async {
    emit(const BlocState.loading());
    
    final result = await _getItemsUseCase.execute();
    
    emit(result.fold(
      (failure) => BlocState.error(failure.message),
      (items) => BlocState.loaded(items),
    ));
  }
